## 🏗️ Technical Implementation

### Frame Module Example
```frame
module MathUtils {
    var PI = 3.14159
    
    fn add(a, b) {
        return a + b
    }
    
    fn multiply(a, b) {
        return a * b
    }
}
```

### Generated TypeScript
```typescript
export namespace MathUtils {
    export let PI: any = 3.14159;

    export function add(a: any, b: any): any {
        return (a + b);
    }

    export function multiply(a: any, b: any): any {
        return (a * b);
    }
}
```
